local OriginalLighting = {}
local SavedAtmospheres = {}
local SavedBlooms = {}

local function SaveOriginalLighting()
    OriginalLighting.ClockTime = Lighting.ClockTime
    OriginalLighting.Ambient = Lighting.Ambient
    OriginalLighting.Brightness = Lighting.Brightness
    OriginalLighting.ColorShift_Bottom = Lighting.ColorShift_Bottom
    OriginalLighting.ColorShift_Top = Lighting.ColorShift_Top
    OriginalLighting.EnvironmentDiffuseScale = Lighting.EnvironmentDiffuseScale
    OriginalLighting.EnvironmentSpecularScale = Lighting.EnvironmentSpecularScale
    OriginalLighting.GlobalShadows = Lighting.GlobalShadows
    OriginalLighting.OutdoorAmbient = Lighting.OutdoorAmbient
    OriginalLighting.ShadowSoftness = Lighting.ShadowSoftness
    OriginalLighting.ExposureCompensation = Lighting.ExposureCompensation
    
    SavedAtmospheres = {}
    SavedBlooms = {}
    for _, child in ipairs(Lighting:GetChildren()) do
        if child:IsA('Atmosphere') then
            local clone = child:Clone()
            clone.Parent = nil
            table.insert(SavedAtmospheres, {original = child, clone = clone})
        elseif child:IsA('BloomEffect') then
            local clone = child:Clone()
            clone.Parent = nil
            table.insert(SavedBlooms, {original = child, clone = clone})
        end
    end
end

local function RestoreLighting()
    pcall(function()
        local akihaAtmos = Lighting:FindFirstChild("AkihaAtmosphere")
        local akihaBloom = Lighting:FindFirstChild("AkihaBloom")
        if akihaAtmos then akihaAtmos:Destroy() end
        if akihaBloom then akihaBloom:Destroy() end
    end)
    
    for prop, val in pairs(OriginalLighting) do
        pcall(function() Lighting[prop] = val end)
    end
    
    for _, child in ipairs(Lighting:GetChildren()) do
        if child:IsA('Atmosphere') or child:IsA('BloomEffect') then
            child.Enabled = true
        end
    end
end

local JumpDebounce = false
local IsGliding = false
local CurrentBodyGyro = nil
local JumpConnection = nil
local OriginalGravity = workspace.Gravity

local function CleanupJumpSystem()
    if CurrentBodyGyro then
        pcall(function() CurrentBodyGyro:Destroy() end)
        CurrentBodyGyro = nil
    end
    IsGliding = false
    JumpDebounce = false
end

local function SetupCustomJumpSystem(char)
    if not CustomJumpEnabled then return end
    if not char then return end
    
    local hum = char:FindFirstChildOfClass("Humanoid")
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hum or not hrp then return end
    
    OriginalGravity = workspace.Gravity
    workspace.Gravity = 80
    hum.JumpPower = 25
    hum.JumpHeight = 7.2
    
    if JumpConnection then
        pcall(function() JumpConnection:Disconnect() end)
    end
    
    JumpConnection = UserInputService.JumpRequest:Connect(function()
        if not CustomJumpEnabled then return end
        if JumpDebounce then return end
        
        JumpDebounce = true
        IsGliding = true
        
        local bodyGyro = Instance.new('BodyGyro', hrp)
        bodyGyro.Name = 'AkihaJumpGyro'
        bodyGyro.MaxTorque = Vector3.new(1, 1, 1) * 10^15
        bodyGyro.D = 4500
        bodyGyro.P = 15500
        CurrentBodyGyro = bodyGyro
        
        hum:ChangeState(Enum.HumanoidStateType.Flying)
        bodyGyro.CFrame = hrp.CFrame * CFrame.Angles(math.rad(-45), 0, 0)
        
        local wasAutoRotate = hum.AutoRotate
        hum.AutoRotate = false
        
        local Anim = Instance.new("Animation")
        Anim.AnimationId = "rbxassetid://121032789756540"
        local animTrack
        pcall(function()
            animTrack = hum:LoadAnimation(Anim)
            animTrack:Play(0.3)
            animTrack.Priority = Enum.AnimationPriority.Action
            animTrack:AdjustSpeed(1)
            animTrack.TimePosition = 5
        end)
        
        task.spawn(function()
            local startTime = tick()
            while CurrentBodyGyro and CurrentBodyGyro.Parent == hrp and (tick() - startTime) < 8 do
                bodyGyro.D = 1500 - math.sin((tick() - startTime) * 10) * 1500
                task.wait(0.01)
            end
        end)
        
        task.delay(0.2, function()
            JumpDebounce = false
            if animTrack then
                animTrack:AdjustSpeed(0.2)
            end
            
            repeat task.wait() until not char or not char.Parent or hum.FloorMaterial ~= Enum.Material.Air or not CustomJumpEnabled
            
            CleanupJumpSystem()
            hum.AutoRotate = wasAutoRotate
            
            if animTrack then
                animTrack:Stop()
            end
            
            local landAnim = Instance.new("Animation")
            landAnim.AnimationId = "rbxassetid://18462892217"
            pcall(function()
                local landTrack = hum:LoadAnimation(landAnim)
                landTrack:Play(0.1)
                landTrack.Priority = Enum.AnimationPriority.Action
                landTrack:AdjustSpeed(1.4)
                task.wait(0.4)
                landTrack:Stop(0.3)
            end)
        end)
    end)
    
    return JumpConnection
end

local function DisableCustomJump()
    if JumpConnection then
        pcall(function() JumpConnection:Disconnect() end)
        JumpConnection = nil
    end
    CleanupJumpSystem()
    workspace.Gravity = OriginalGravity or 196.2
end

local function FindNearestEnemy()
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    
    local nearestEnemy = nil
    local nearestDist = math.huge
    
    for _, part in ipairs(workspace.Live:GetChildren()) do
        if part:IsA('Model') and part ~= char and part:FindFirstChild('HumanoidRootPart') then
            local enemyHum = part:FindFirstChildOfClass("Humanoid")
            if enemyHum and enemyHum.Health > 0 then
                local dist = (part.HumanoidRootPart.Position - hrp.Position).Magnitude
                if dist < nearestDist and dist < 20 then
                    nearestDist = dist
                    nearestEnemy = part
                end
            end
        end
    end
    
    return nearestEnemy
end

local function DoHitEffect(doLightning, dir, amount)
    DoCustomShake()
    
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    local target = EnemyTarget or FindNearestEnemy()
    
    if doLightning and target and target:FindFirstChild("HumanoidRootPart") then
        CreateLightningEffect(hrp.Position, target.HumanoidRootPart.Position + Vector3.new(0, dir or 15, 0))
    end
    
    local p = Instance.new('Part')
    p.Parent = workspace:FindFirstChild("Thrown") or workspace
    p.Material = Enum.Material.Neon
    p.Size = Vector3.new(1, 1, 1)
    p.Anchored = true
    p.CanCollide = false
    p.Color = Color3.new(1, 0, 0)
    
    local msh = Instance.new("SpecialMesh", p)
    msh.MeshType = "Sphere"
    msh.Scale = Vector3.new(1, 1, 15)
    
    if target and target:FindFirstChild("HumanoidRootPart") then
        task.spawn(function()
            for i = 1, 55 do
                if not target or not target:FindFirstChild("HumanoidRootPart") then break end
                p.CFrame = target.HumanoidRootPart.CFrame * CFrame.new(0, 0, -2 + i / 3)
                task.wait(0.01)
            end
        end)
    end
    
    game:GetService("Debris"):AddItem(p, 1)
    TweenService:Create(msh, TweenInfo.new(0.3), { Scale = Vector3.new(0, 0, 15) }):Play()
    
    for i = 1, (amount or 5) do
        local sp = Instance.new('Part')
        sp.Parent = workspace:FindFirstChild("Thrown") or workspace
        sp.Material = Enum.Material.Neon
        sp.Size = Vector3.new(1, 1, 1)
        sp.Anchored = true
        sp.CanCollide = false
        sp.Color = Color3.new(1, 0, 0)
        
        local sm = Instance.new("SpecialMesh", sp)
        sm.MeshType = "Sphere"
        sm.Scale = Vector3.new(3, 1, 3)
        
        if target and target:FindFirstChild("HumanoidRootPart") then
            task.spawn(function()
                for j = 1, 35 do
                    if not target or not target:FindFirstChild("HumanoidRootPart") then break end
                    sp.CFrame = target.HumanoidRootPart.CFrame * CFrame.Angles(math.random(-90, 90), math.random(-90, 90), math.random(-90, 90))
                    task.wait(0.01)
                end
            end)
        end
        
        game:GetService("Debris"):AddItem(sp, 1.5)
        TweenService:Create(sm, TweenInfo.new(0.5), { Scale = Vector3.new(0, 5 + i * 4, 0) }):Play()
    end
end

RunService.Heartbeat:Connect(function()
    if not AutoBlockEnabled then
        if AutoBlockIsBlocking then
            StopAutoBlock()
            AutoBlockIsBlocking = false
        end
        return
    end
    
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    
    local myPosition = char.HumanoidRootPart.Position
    local closeRangeTarget = nil
    local closestCloseRange = math.huge
    local longRangeTarget = nil
    local closestLongRange = math.huge
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local targetChar = player.Character
            if targetChar and targetChar:FindFirstChild("HumanoidRootPart") then
                local distance = (myPosition - targetChar.HumanoidRootPart.Position).Magnitude
                
                if distance < AutoBlockCloseRange and distance < closestCloseRange then
                    closestCloseRange = distance
                    closeRangeTarget = player
                end
                
                if distance < AutoBlockLongRange and distance < closestLongRange then
                    closestLongRange = distance
                    longRangeTarget = player
                end
            end
        end
    end
    
    local shouldBlock = false
    
    if longRangeTarget then
        local targetHum = longRangeTarget.Character:FindFirstChildOfClass("Humanoid")
        if targetHum and IsPlayingEnemyAnimation(targetHum, LongRangeAnimations) then
            shouldBlock = true
        end
    end
    
    if not shouldBlock and closeRangeTarget then
        local targetHum = closeRangeTarget.Character:FindFirstChildOfClass("Humanoid")
        if targetHum and IsPlayingEnemyAnimation(targetHum, CloseRangeAnimations) then
            shouldBlock = true
        end
    end
    
    if shouldBlock then
        if not AutoBlockIsBlocking then
            AutoBlockIsBlocking = true
            StartAutoBlock()
        end
    elseif AutoBlockIsBlocking then
        AutoBlockIsBlocking = false
        StopAutoBlock()
        if M1AfterBlockEnabled then
            StartM1Attack()
            task.delay(0.11, StopM1Attack)
        end
    end
end)

local EnemyTarget = nil

local function SetupHitDetection(char)
    if not char then return end

    char:GetAttributeChangedSignal("LastDamageDone"):Connect(function()
        if not AkihaEnabled then return end

        LastHitTime = os.clock()

        for _, part in ipairs(workspace.Live:GetChildren()) do
            if part:IsA("Model") and part:FindFirstChild("HumanoidRootPart") then
                local You = string.sub(tostring(char:GetAttribute("LastDamageDone") or ""), 1, 13)
                local Them = string.sub(tostring(part:GetAttribute("LastDamage") or ""), 1, 13)

                if You == Them then
                    EnemyTarget = part
                    local enemyHum = part:FindFirstChildOfClass("Humanoid")

                    if enemyHum then
                        local newHealth = enemyHum.Health

                        if EnemyHealth and newHealth < EnemyHealth then
                            local damageDone = EnemyHealth - newHealth
                            ComboDamage = ComboDamage + damageDone
                            ComboHits = ComboHits + 1
                            UpdateComboCounter(ComboHits, ComboDamage)

                            DoCustomShake()

                            local enemyHrp = part:FindFirstChild("HumanoidRootPart")
                            local myHrp = char:FindFirstChild("HumanoidRootPart")

                            if enemyHrp and myHrp then
                                CreateBloodParticle(enemyHrp)
                                CreateLightningEffect(myHrp.Position, enemyHrp.Position)
                            end
                        end

                        if newHealth <= 0 and not BloodHeatActive then
                            ActivateBloodHeat()

                            if AutoRoastEnabled and RoastMessages and #RoastMessages > 0 then
                                local roastMsg = RoastMessages[math.random(1, #RoastMessages)]

                                task.spawn(function()
                                    local ReplicatedStorage = game:GetService("ReplicatedStorage")
                                    local ChatEvents = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")

                                    if ChatEvents then
                                        local SayMessage = ChatEvents:FindFirstChild("SayMessageRequest")
                                        if SayMessage then
                                            SayMessage:FireServer(roastMsg, "All")
                                        end
                                    end
                                end)
                            end
                        end

                        EnemyHealth = newHealth
                        OldEnemyHealth = newHealth
                    end
                end
            end
        end
    end)
end

local AkihaConnections = {}

    for _, conn in ipairs(AkihaConnections) do
        pcall(function() conn:Disconnect() end)
    end
    AkihaConnections = {}
end

local function SetupAkihaFeatures(char)
    if not char then return end
    CleanupAkihaConnections()
    
    local hum = char:WaitForChild("Humanoid")
    local hrp = char:WaitForChild("HumanoidRootPart")
    
    table.insert(AkihaConnections, SetupHitDetection(char))
    
    if CustomJumpEnabled then
        local jumpConn = SetupCustomJumpSystem(char)
        if jumpConn then
            table.insert(AkihaConnections, jumpConn)
        end
        
        table.insert(AkihaConnections, hum:GetPropertyChangedSignal('JumpPower'):Connect(function()
            if CustomJumpEnabled and hum.JumpPower ~= 25 then
                hum.JumpPower = 25
            end
        end))
        
        table.insert(AkihaConnections, hum:GetPropertyChangedSignal('PlatformStand'):Connect(function()
            if hum.PlatformStand == true then
                hum.PlatformStand = false
            end
        end))
    end
    
    table.insert(AkihaConnections, hum.AnimationPlayed:Connect(function(animTrack)
        if not CustomAnimationsEnabled then return end
        local anim = animTrack.Animation
        if not anim then return end
        
        if M1AnimIds[anim.AnimationId] then
            HandleM1Animation(animTrack, char)
        end
        
        for _, moveAnimId in pairs(MoveAnimIds) do
            if anim.AnimationId == moveAnimId then
                HandleMoveAnimation(animTrack, char)
                break
            end
        end
    end))
end


local AnimatorConnections = {}

local function DisconnectAnimatorListeners()
    for _, conn in ipairs(AnimatorConnections) do
        if conn then pcall(function() conn:Disconnect() end) end
    end
    AnimatorConnections = {}
end

local function AttachAnimatorListeners(animator)
    if not animator then return end
    DisconnectAnimatorListeners()
    
    table.insert(AnimatorConnections, animator.AnimationPlayed:Connect(function(animTrack)
        if not AutoVoidEnabled or IsVoiding then return end
        local animation = animTrack.Animation
        if not animation then return end
        for _, animData in ipairs(VoidAnimations) do
            local animId = animation.AnimationId:gsub("rbxassetid://", "")
            if animId == animData.id then
                IsVoiding = true
                local char = LocalPlayer.Character
                local hum = char and char:FindFirstChildOfClass("Humanoid")
                local rootPart = char and char:FindFirstChild("HumanoidRootPart")
                if not char or not char.Parent or not hum or hum.Health <= 0 or not rootPart then
                    IsVoiding = false
                    return
                end
                local oldCFrame = rootPart.CFrame
                task.wait(animData.timewait)
                char = LocalPlayer.Character
                hum = char and char:FindFirstChildOfClass("Humanoid")
                rootPart = char and char:FindFirstChild("HumanoidRootPart")
                if char and char.Parent and AutoVoidEnabled and hum and hum.Health > 0 and rootPart then
                    rootPart.CFrame = CFrame.new(TeleportPosition)
                    animTrack.Stopped:Wait()
                    if TpBackEnabled then
                        char = LocalPlayer.Character
                        hum = char and char:FindFirstChildOfClass("Humanoid")
                        rootPart = char and char:FindFirstChild("HumanoidRootPart")
                        if char and char.Parent and AutoVoidEnabled and hum and hum.Health > 0 and rootPart then
                            rootPart.CFrame = oldCFrame
                        end
                    end
                end
                IsVoiding = false
                break
            end
        end
    end))
    
    table.insert(AnimatorConnections, animator.AnimationPlayed:Connect(function(animTrack)
        if AutoWhirlwindDunk and animTrack.Animation then
            local animId = animTrack.Animation.AnimationId
            if animId == WhirlwindAnimId or animId == "rbxassetid://12296113986" then
                task.wait(1.2)
                local char = LocalPlayer.Character
                local rootPart = char and char:FindFirstChild("HumanoidRootPart")
                if rootPart then
                    local pos = rootPart.Position
                    rootPart.CFrame = CFrame.new(pos.X, pos.Y + 100, pos.Z)
                end
            end
        end
    end))
end

local function CreateUltEsp(player, character)
    if not character then return end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local head = character:FindFirstChild("Head")
    if not humanoid or not head then return end
    if UltEspData[player] then UltEspData[player][1]() end
    local espText = Drawing.new("Text")
    espText.Visible = false
    espText.Center = true
    espText.Outline = true
    espText.Font = Drawing.Fonts.UI
    espText.Size = 14
    espText.Color = Color3.fromRGB(255, 255, 0)
    local function GetUltText()
        local liveChar = workspace.Live:FindFirstChild(player.Name)
        if liveChar then
            local ultBar = liveChar:GetAttribute("UltimateBar") or 0
            return string.format("[ULT: %.0f%%]", ultBar * 100)
        end
        return "[ULT: 0%]"
    end
    local connections = {}
    local function Cleanup()
        espText:Remove()
        for _, conn in ipairs(connections) do conn:Disconnect() end
        UltEspData[player] = nil
    end
    table.insert(connections, character.AncestryChanged:Connect(function(_, parent)
        if not parent then Cleanup() end
    end))
    table.insert(connections, humanoid.Died:Connect(Cleanup))
    table.insert(connections, RunService.RenderStepped:Connect(function()
        if EspUltBar and head and head.Parent then
            local screenPos, onScreen = workspace.CurrentCamera:WorldToViewportPoint(head.Position)
            if onScreen then
                espText.Position = Vector2.new(screenPos.X, screenPos.Y - 42)
                espText.Text = GetUltText()
                espText.Visible = true
            else
                espText.Visible = false
            end
        else
            espText.Visible = false
        end
    end))
    UltEspData[player] = {Cleanup}
end

local function ClearAllUltEsp()
    for _, data in pairs(UltEspData) do if data[1] then data[1]() end end
    UltEspData = {}
end

local function SetupUltEspForPlayer(player)

local espEnabled = false
local chamsEnabled = false
local rainbowCycleSpeed = 1
local espInstances = {}

local function createESP(player)
    if player == LocalPlayer then return end
    local char = player.Character
    if not char then return end
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "ESP"
    billboard.AlwaysOnTop = true
    billboard.Size = UDim2.new(0, 200, 0, 50)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.Adornee = char:FindFirstChild("Head")
    billboard.Parent = char:FindFirstChild("Head")
    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1, 0, 0.5, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = player.Name
    nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    nameLabel.TextSize = 14
    nameLabel.Parent = billboard
    local healthLabel = Instance.new("TextLabel")
    healthLabel.Size = UDim2.new(1, 0, 0.5, 0)
    healthLabel.Position = UDim2.new(0, 0, 0.5, 0)
    healthLabel.BackgroundTransparency = 1
    healthLabel.Text = ""
    healthLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
    healthLabel.TextSize = 14
    healthLabel.Parent = billboard
    return {billboard = billboard, healthLabel = healthLabel, chams = {}}
end

local function removeESP(player)
    if espInstances[player] then
        if espInstances[player].billboard then espInstances[player].billboard:Destroy() end
        for _, cham in pairs(espInstances[player].chams or {}) do cham:Destroy() end
        espInstances[player] = nil
    end
end

local function updateESP()
    for player, data in pairs(espInstances) do
        local char = player.Character
        if char and data.healthLabel then
            local humanoid = char:FindFirstChild("Humanoid")
            if humanoid then data.healthLabel.Text = tostring(math.floor(humanoid.Health)) end
        end
    end
end

local function updateRainbowChams()
    local hue = (tick() * rainbowCycleSpeed) % 1
    local color = Color3.fromHSV(hue, 1, 1)
    for _, data in pairs(espInstances) do
        for _, cham in pairs(data.chams or {}) do
            cham.FillColor = color
            cham.OutlineColor = color
        end
    end
end

FarmObj.isTargetable = function(char)
    if char:GetAttribute("Cutscene") or char:GetAttribute("Invincible") or char:FindFirstChild("Cutscene") then
        return false
    end
    FarmObj.tempHum = char:FindFirstChild("Humanoid")
    return FarmObj.tempHum and FarmObj.tempHum.Health > 0
end

FarmObj.getNewTarget = function()
    if not FarmObj.liveFolder then return nil end
    
    FarmObj.tiers = {
        ["Low Health"] = {},
        ["AFK Players"] = {},
        ["Killstreaks (10+)"] = {},
        ["Highest Kills (Most Experienced)"] = {},
        ["Oldest Players (First Player that Joined)"] = {}
    }

    for _, char in ipairs(FarmObj.liveFolder:GetChildren()) do
        FarmObj.p = FarmObj.Players:GetPlayerFromCharacter(char)
        FarmObj.hum = char:FindFirstChild("Humanoid")
        
        if FarmObj.p and FarmObj.p ~= FarmObj.player and FarmObj.isTargetable(char) and FarmObj.hum then
            FarmObj.hpPct = FarmObj.hum.Health / FarmObj.hum.MaxHealth
            FarmObj.strk = FarmObj.getStreak(char)

            if FarmObj.hpPct <= 0.35 then table.insert(FarmObj.tiers["Low Health"], char) end
            if char:FindFirstChild("HumanoidRootPart") and char.HumanoidRootPart.Velocity.Magnitude < 1.5 then table.insert(FarmObj.tiers["AFK Players"], char) end
            if FarmObj.strk >= 10 then table.insert(FarmObj.tiers["Killstreaks (10+)"], char) end
            if (FarmObj.p:GetAttribute("Kills") or 0) > 50 then table.insert(FarmObj.tiers["Highest Kills (Most Experienced)"], char) end
            if FarmObj.p.UserId < 800000000 then table.insert(FarmObj.tiers["Oldest Players (First Player that Joined)"], char) end
        end
    end

    if FarmObj.verifyTargetEnabled and #FarmObj.PriorityList > 0 then
        for _, priorityName in ipairs(FarmObj.PriorityList) do
            FarmObj.possibleTargets = FarmObj.tiers[priorityName]
            if FarmObj.possibleTargets and #FarmObj.possibleTargets > 0 then
                return FarmObj.possibleTargets[math.random(1, #FarmObj.possibleTargets)]
            end
        end
    end

    for _, fallbackTier in pairs(FarmObj.tiers) do
        if #fallbackTier > 0 then
            return fallbackTier[math.random(1, #fallbackTier)]
        end
    end
    
    return nil
end

task.spawn(function()
    while true do
        task.wait(0.01)
        if not FarmObj.autoFarmEnabled then continue end
        FarmObj.char = FarmObj.getLiveChar(FarmObj.player.Name)
        if not FarmObj.char then continue end
        FarmObj.hmo = FarmObj.char:FindFirstChild("Humanoid")
        FarmObj.rp = FarmObj.char:FindFirstChild("HumanoidRootPart")
        
        if not FarmObj.hmo or not FarmObj.rp or FarmObj.hmo.Health <= 0 then 
            FarmObj.isHealing = false
            FarmObj.manageHealPlatform(false)
            continue 
        end
        
        FarmObj.currHP = (FarmObj.hmo.Health / FarmObj.hmo.MaxHealth) * 100
        FarmObj.currStrk = FarmObj.getStreak(FarmObj.char)
        
        FarmObj.shouldResetHealth = (FarmObj.settingsApplied and FarmObj.currHP <= FarmObj.lowHealthThreshold)
        FarmObj.shouldResetStreak = (FarmObj.settingsApplied and FarmObj.streakLimit > 0 and FarmObj.currStrk >= FarmObj.streakLimit)

        if (FarmObj.shouldResetHealth or FarmObj.shouldResetStreak) and not FarmObj.isHealing then 
            FarmObj.isHealing = true
            FarmObj.skyPos = Vector3.new(FarmObj.rp.Position.X, 700, FarmObj.rp.Position.Z)
            FarmObj.rp.CFrame = CFrame.new(FarmObj.skyPos)
            FarmObj.manageHealPlatform(true, FarmObj.skyPos)
            
            WindUI:Notify({
                Title = "Safety Reset",
                Content = "Limit reached. Resetting in 20s.",
                Duration = 5,
                Icon = "shield-check",
            })

            if FarmObj.hasPlayerTrashCan() then
                FarmObj.char.Communicate:FireServer({Goal = "LeftClick", Mobile = true})
                task.wait(0.05)
                FarmObj.char.Communicate:FireServer({Goal = "LeftClickRelease", Mobile = true})
            end
            task.wait(20) 
            if FarmObj.hmo and FarmObj.hmo.Health > 0 then FarmObj.hmo.Health = 0 end
        end

        if FarmObj.isHealing then continue end
        
        if not FarmObj.target or not FarmObj.target.Parent or not FarmObj.isTargetable(FarmObj.target) or FarmObj.throwCount >= 15 then
            FarmObj.target = FarmObj.getNewTarget()
            FarmObj.throwCount = 0
        end

        if FarmObj.hasPlayerTrashCan() then
            if FarmObj.target then
                FarmObj.tHrp = FarmObj.target:FindFirstChild("HumanoidRootPart")
                if FarmObj.tHrp then
                    FarmObj.dist = FarmObj.settingsApplied and FarmObj.targetDistance or 12
                    FarmObj.rp.CFrame = CFrame.new((FarmObj.tHrp.CFrame * CFrame.new(0, 0, FarmObj.dist)).Position, FarmObj.tHrp.Position)
                    task.wait(0.1)
                    FarmObj.thrStart = tick()
                    repeat
                        FarmObj.char.Communicate:FireServer({Goal = "LeftClick", Mobile = true})
                        task.wait(0.1)
                        FarmObj.char.Communicate:FireServer({Goal = "LeftClickRelease", Mobile = true})
                        task.wait(0.05)
                    until not FarmObj.hasPlayerTrashCan() or tick() - FarmObj.thrStart > 1.5
                    FarmObj.throwCount = FarmObj.throwCount + 1
                end
            end
        else
            FarmObj.trshP, FarmObj.trshM = nil, nil
            for _, tc in ipairs(FarmObj.trashFolder:GetChildren()) do
                if tc:IsA("Model") and not tc:GetAttribute("Broken") then
                    FarmObj.trshP = tc.PrimaryPart or tc:FindFirstChildWhichIsA("BasePart")
                    FarmObj.trshM = tc
                    if FarmObj.trshP then break end
                end
            end
            
            if FarmObj.trshP then
                FarmObj.pStart = tick()
                FarmObj.lTp = 0
                repeat
                    if tick() - FarmObj.lTp >= 0.35 then
                        FarmObj.rp.CFrame = CFrame.new(FarmObj.trshP.Position + Vector3.new(1.2, 0, 1.2), FarmObj.trshP.Position)
                        FarmObj.lTp = tick()
                    end
                    FarmObj.char.Communicate:FireServer({Goal = "LeftClick", Mobile = true})
                    task.wait(0.08)
                    FarmObj.char.Communicate:FireServer({Goal = "LeftClickRelease", Mobile = true})
                    task.wait(0.08)
                until FarmObj.hasPlayerTrashCan() or tick() - FarmObj.pStart > 4 or not FarmObj.autoFarmEnabled
                
                if not FarmObj.hasPlayerTrashCan() and tick() - FarmObj.pStart > 4 then
                    FarmObj.trshM:SetAttribute("Broken", true)
                elseif FarmObj.hasPlayerTrashCan() and FarmObj.trshM then 
                    FarmObj.trshM:SetAttribute("Broken", true) 
                end
            else
                WindUI:Notify({
                    Title = "Waiting for Spawn",
                    Content = "No trashcans left. Teleporting to safety.",
                    Duration = 3,
                    Icon = "clock",
                })
                
                FarmObj.wPos = Vector3.new(FarmObj.rp.Position.X, 700, FarmObj.rp.Position.Z)
                FarmObj.rp.CFrame = CFrame.new(FarmObj.wPos)
                FarmObj.manageHealPlatform(true, FarmObj.wPos)
                
                repeat 
                    task.wait(0.5) 
                    for _, tc in ipairs(FarmObj.trashFolder:GetChildren()) do
                        if tc:IsA("Model") and not tc:GetAttribute("Broken") then
                            FarmObj.foundPart = tc.PrimaryPart or tc:FindFirstChildWhichIsA("BasePart")
                            if FarmObj.foundPart then FarmObj.trshP = FarmObj.foundPart break end
                        end
                    end
                until FarmObj.trshP or not FarmObj.autoFarmEnabled or FarmObj.hmo.Health <= 0
                
                FarmObj.manageHealPlatform(false)
            end
        end
    end
end)

function tweenForward(root, distance, duration)
    local start = root.CFrame
    local goal = start + start.LookVector * distance
    local elapsed = 0
    while elapsed < duration do
        elapsed = elapsed + task.wait()
        local alpha = math.clamp(elapsed/duration,0,1)
        root.CFrame = start:Lerp(goal, alpha)
    end
end

function SetTargetCollideState(model, state)
    for _, p in ipairs(model:GetDescendants()) do
        if p:IsA("BasePart") then
            p.CanCollide = state
        end
    end
end

function StopFollow()
    if followConnection then
        followConnection:Disconnect()
        followConnection = nil
    end

    char = LocalPlayer.Character
    if char and char:FindFirstChild("Humanoid") then
        char.Humanoid:Move(Vector3.zero, false)
    end

    if currentTarget then
        SetTargetCollideState(currentTarget, true)
    end

    currentTarget = nil
    isFollowing = false
end

function FollowTargetModel()
    char = LocalPlayer.Character
    if not char then return end

    hrp = char:WaitForChild("HumanoidRootPart")
    hum = char:WaitForChild("Humanoid")

    hum.WalkSpeed = 30

    if followConnection then
        followConnection:Disconnect()
    end

    followConnection = RunService.Heartbeat:Connect(function()
        if not isFollowing or not currentTarget then return end
        if not currentTarget:FindFirstChild("HumanoidRootPart") or currentTarget.Humanoid.Health <= 0 then return end

        tHRP = currentTarget.HumanoidRootPart
        behind = tHRP.Position - tHRP.CFrame.LookVector * 2

        hrp.CFrame = CFrame.new(hrp.Position, tHRP.Position)

        if (behind - hrp.Position).Magnitude > 2 then
            hum:MoveTo(behind)
        end
    end)
end

function JumpSmooth(hrp)
    back = -hrp.CFrame.LookVector * 4

    bv = Instance.new("BodyVelocity")
    bv.Velocity = Vector3.new(back.X, 30, back.Z)
    bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bv.P = 5000
    bv.Parent = hrp

    game.Debris:AddItem(bv, 0.25)
end

function WatchForAnimation(hum)
    if animConnection then
        animConnection:Disconnect()
        animConnection = nil
    end

    animConnection = hum.AnimationPlayed:Connect(function(track)
        if not isEnabled then return end

        if track.Animation.AnimationId == animationIdToDetect then
            if tick() - lastUsed < cooldownTime then return end
            lastUsed = tick()

            char = hum.Parent
            hrp = char:FindFirstChild("HumanoidRootPart")
            if hrp then
                JumpSmooth(hrp)
            end

            task.wait(0.2)

            comm = char:FindFirstChild("Communicate")
            if comm then
                comm:FireServer({Dash = Enum.KeyCode.W, Key = Enum.KeyCode.Q, Goal = "KeyPress"})
            end

            currentTarget = GetNearestLiveInFront()

            if currentTarget then
                SetTargetCollideState(currentTarget, false)
                isFollowing = true
                task.spawn(FollowTargetModel)
                task.delay(1, StopFollow)
            end
        end
    end)
end

function wiggle()
	start = tick()
	dir = 1
	while tick() - start <= WIGGLE_DURATION do
		if hrp then
			pos = hrp.Position
			rot = CFrame.Angles(0, math.rad(WIGGLE_DEGREES * dir), 0)
			hrp.CFrame = CFrame.new(pos) * rot
			dir = dir * -1
		end
		task.wait(WIGGLE_SPEED)
	end
end

function getNearestTarget()
	closest = nil
	shortest = 20
	lp = game.Players.LocalPlayer
	char = lp.Character
	hrp = char and char:FindFirstChild("HumanoidRootPart")

	if not hrp then return nil end

	for _,plr in ipairs(game.Players:GetPlayers()) do
		if plr ~= lp and plr.Character then
			r = plr.Character:FindFirstChild("HumanoidRootPart")
			h = plr.Character:FindFirstChildOfClass("Humanoid")
			if r and h and h.Health > 0 then
				d = (r.Position - hrp.Position).Magnitude
				if d < shortest then
					shortest = d
					closest = r
				end
			end
		end
	end

	for _,model in ipairs(workspace:GetDescendants()) do
		if model:IsA("Model") and model ~= char then
			r = model:FindFirstChild("HumanoidRootPart")
			h = model:FindFirstChildOfClass("Humanoid")
			if r and h and h.Health > 0 then
				d = (r.Position - hrp.Position).Magnitude
				if d < shortest then
					shortest = d
					closest = r
				end
			end
		end
	end

	return closest
end

function lockToTarget(target)
	if not target then return end

	lp = game.Players.LocalPlayer
	char = lp.Character
	hum = char and char:FindFirstChildOfClass("Humanoid")
	hrp = char and char:FindFirstChild("HumanoidRootPart")

	if not hum or not hrp then return end

	task.wait(LOCK_DELAY)
	startTime = tick()
	hum:ChangeState(Enum.HumanoidStateType.Physics)

	conn = game:GetService("RunService").Heartbeat:Connect(function()
		if not hrp or not target or not target.Parent then
			conn:Disconnect()
			return
		end

		if tick() - startTime >= LOCK_TIME then
			conn:Disconnect()
			hum:ChangeState(Enum.HumanoidStateType.GettingUp)
			return
		end

		pos = target.Position + Vector3.new(0, LOCK_HEIGHT, 0)
		cf = CFrame.lookAt(pos, pos + target.CFrame.LookVector) * CFrame.Angles(math.rad(-90),0,0)
		hrp.CFrame = cf
	end)
end

function executeTech()
	if busy then return end
	busy = true

	target = getNearestTarget()
	if not target then busy = false return end

	lp = game.Players.LocalPlayer
	char = lp.Character
	hrp = char and char:FindFirstChild("HumanoidRootPart")

	if not hrp then busy = false return end

	hrp.CFrame = CFrame.new(target.Position + Vector3.new(0, LOCK_HEIGHT, 0))
	lockToTarget(target)
	pressQ()
	task.wait(POST_Q_DELAY)
	wiggle()

	task.delay(LOCK_TIME + 0.1, function()
		busy = false
	end)
end

	if camConn then

function startCamlock()
	stopCamlock()
	
	camConn = RunService.RenderStepped:Connect(function()
		if not enabled then return end
		
		target = getNearestTarget()
		
		if target and target:FindFirstChild("HumanoidRootPart") then
			targetPos = target.HumanoidRootPart.Position
			
			offset = (cam.CFrame.Position - lp.Character.HumanoidRootPart.Position).Magnitude
			
			dir = (targetPos - lp.Character.HumanoidRootPart.Position).Unit
			
			newCamPos = lp.Character.HumanoidRootPart.Position - dir * offset + Vector3.new(0, 2, 0)

			cam.CFrame = cam.CFrame:Lerp(
				CFrame.new(newCamPos, targetPos),
				alpha
			)
		end
	end)

local function ToggleBlock(state)

local function WaitForAttribute(instance, attribute, timeout)
    local start = os.clock()
    while instance:GetAttribute(attribute) == nil do
        if os.clock() - start > (timeout or 10) then return nil end
        task.wait()
    end
    return instance:GetAttribute(attribute)
end

local function ReplaceAnimations(animConfig)
    GetCharacter():WaitForChild("Humanoid").AnimationPlayed:Connect(function(track)
        local config = animConfig[track.Animation.AnimationId]
        if not config then return end

        if typeof(config.Callback) == "function" then
            task.spawn(config.Callback, track)
        end

        track:AdjustSpeed(0) 

        if config.Kill then
            track:Stop()
            return 
        end

        if not config.UpdateToAnimation then track:Stop() end

        local newTrack = PlayAnimation(config.Animation)
        if config.Speed then newTrack:AdjustSpeed(config.Speed) end
        if config.Priority then newTrack.Priority = config.Priority end
        
        if config.StopOnLength then
            task.delay(track.Length, function() track:Stop() end)
        end
    end)
end

function fadeOutParticles(obj, duration)
	for _, c in ipairs(obj:GetChildren()) do
		if c:IsA("ParticleEmitter") then
			c.Enabled = true
			startTime = tick()
			conn = RunService.RenderStepped:Connect(function()
				elapsed = tick() - startTime
				alpha = math.clamp(1 - (elapsed / duration), 0, 1)
				c.Transparency = NumberSequence.new(alpha)
				if alpha <= 0 then
					conn:Disconnect()
					obj:Destroy()
				end
			end)
		end
	end
end

function setupVFX(path, emitAmount, enable, fadeDuration)

			c:Emit(emitAmount)
			c.Enabled = enable
		end
	end
	if fadeDuration then
		fadeOutParticles(obj, fadeDuration)
	end
end

function isBW(color)
	return (color.R <= 0.01 and color.G <= 0.01 and color.B <= 0.01)
		or (color.R >= 0.99 and color.G >= 0.99 and color.B >= 0.99)
end

function colorSequenceIsBW(colorSeq)
	for _, keypoint in ipairs(colorSeq.Keypoints) do
		if not isBW(keypoint.Value) then
			return false
		end
	end
	return true
end

function emit(pe, position)
	if not pe:IsA("ParticleEmitter") then return end
	if pe.Parent.Name == "Attachment" then return end
	if not pe.Color or not colorSequenceIsBW(pe.Color) then return end
	if pe.Size then
		size = pe.Size.Keypoints[1].Value
		if size < 1 then return end
	end
	if pe.Speed and pe.Speed.Max > 60 then return end
	for i = 1, 3 do
		att = Instance.new("Attachment")
		att.Parent = workspace
		offsetX = math.random(-15, 15) / 10
		offsetZ = math.random(-15, 15) / 10
		att.WorldPosition = (position or root.Position) + Vector3.new(offsetX, 0, offsetZ)
		p = pe:Clone()
		p.Parent = att
		p.Enabled = false
		p.Size = NumberSequence.new({NumberSequenceKeypoint.new(0, 5), NumberSequenceKeypoint.new(1, 2)})
		p.Speed = NumberRange.new(pe.Speed.Min * 1.5, pe.Speed.Max * 1.5)
		p.SpreadAngle = Vector2.new(pe.SpreadAngle.X * 1.2, pe.SpreadAngle.Y * 1.2)
		p.Acceleration = pe.Acceleration * 1.2
		p:Emit(1)
		Debris:AddItem(att, 3)
	end
end

		end
	end
end

function bladeCollision(blade1, blade2)
	t1 = blade1:FindFirstChild("ChainPart1")
	t2 = blade2:FindFirstChild("ChainPart1")
	if not t1 or not t2 then return end

	groundLock = { [blade1]=false, [blade2]=false }

	RunService.RenderStepped:Connect(function()
		if not active then return end
		now = tick()
		k = t1:GetDebugId()..t2:GetDebugId()
		if (t1.Position - t2.Position).Magnitude <= 2 then
			if not lastClash[k] or now - lastClash[k] > CLASH_CD then
				spawnClash(t1)
				spawnClash(t2)
				lastClash[k] = now
				groundLock[blade1] = false
				groundLock[blade2] = false
			end
		end

		for _, d in ipairs({
			{tip=t1, blade=blade1, handle=blade1.Handle},
			{tip=t2, blade=blade2, handle=blade2.Handle}
		}) do
			if groundLock[d.blade] then continue end

			rp = RaycastParams.new()
			rp.FilterDescendantsInstances = {char}
			rp.FilterType = Enum.RaycastFilterType.Blacklist

			hit = workspace:Raycast(d.tip.Position, Vector3.new(0,-3,0), rp)
			if hit then
				feet = root.Position - Vector3.new(0, humanoid.HipHeight, 0)
				if (hit.Position - feet).Magnitude <= 4 then
					groundLock[d.blade] = true
					task.spawn(function()
						task.wait(0.3) spawnGround(d.handle)
						task.wait(0.7) spawnGround(d.handle)
						task.wait(1.0) spawnGround(d.handle)
					end)
				end
			end
		end
	end)
end

local function normalize(id)
	id = tostring(id)
	id = id:gsub("http://www.roblox.com/asset/%?id=", "rbxassetid://")
	id = id:gsub("https://assetgame.roblox.com/asset/%?id=", "rbxassetid://")
	return id
end

local function shouldRemove(obj)
	if not fpsEnabled then return false end
	if bannedNames[obj.Name] then return true end
	if obj:IsA("ParticleEmitter") then
		local tex = normalize(obj.Texture)
		if bannedTextures[tex] then return true end
	end
	return false
end

local function CleanWorkspace()
	if not fpsEnabled then return end
	for _, v in ipairs(Workspace:GetDescendants()) do
		if shouldRemove(v) then
			v:Destroy()
		end
	end
end

                        OriginalAssets[v] = {Enabled = v:IsA("PostEffect") and v.Enabled or v.Visible}
                        if v:IsA("Explosion") then v.Visible = false else v.Enabled = false end
                    end
                end)

            OriginalAssets[Lighting] = {GlobalShadows = Lighting.GlobalShadows}
            Lighting.GlobalShadows = false
        else
            for inst, props in pairs(OriginalAssets) do
                if inst and inst.Parent then
                    pcall(function()
                        for name, val in pairs(props) do
                            inst[name] = val
                        end
                    end)
                end
            end
